# Dependências para o ambiente de desenvolvimento e teste do projeto Cortexa

# Framework principal para a escrita e execução de testes unitários.
pytest

# Plugin para pytest que facilita o mocking de objetos e funções,
# essencial para isolar os testes de dependências externas.
pytest-mock

# O SDK da AWS para Python, necessário nos testes para mockar
# as chamadas ao cliente Lambda (boto3.client('lambda')).
boto3

# Adaptador PostgreSQL para Python, usado para conectar ao Neon DB.
# Mesmo que a conexão seja mockada nos testes, o pacote é necessário
# para que o `import psycopg2` no código da aplicação funcione.
psycopg2-binary
